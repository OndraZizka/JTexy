476:     /**
477:      * Converts document in Texy! to (X)HTML code.
478:      *
479:      * @param  string   input text
480:      * @param  bool     is single line?
481:      * @return string   output HTML code
482:      */
483:     public function process($text, $singleLine = FALSE)
484:     {
485:         if ($this->processing) {
486:             throw new InvalidStateException('Processing is in progress yet.');
487:         }
488: 
489:         // initialization
490:         $this->marks = array();
491:         $this->processing = TRUE;
492: 
493:         // speed-up
494:         if (is_array($this->allowedClasses)) $this->_classes = array_flip($this->allowedClasses);
495:         else $this->_classes = $this->allowedClasses;
496: 
497:         if (is_array($this->allowedStyles)) $this->_styles = array_flip($this->allowedStyles);
498:         else $this->_styles = $this->allowedStyles;
499: 
500:         // convert to UTF-8 (and check source encoding)
501:         $text = TexyUtf::toUtf($text, $this->encoding);
502: 
503:         if ($this->removeSoftHyphens) {
504:             $text = str_replace("\xC2\xAD", '', $text);
505:         }
506: 
507:         // standardize line endings and spaces
508:         $text = self::normalize($text);
509: 
510:         // replace tabs with spaces
511:         $this->tabWidth = max(1, (int) $this->tabWidth);
512:         while (strpos($text, "\t") !== FALSE) {
513:             $text = preg_replace_callback('#^(.*)\t#mU', array($this, 'tabCb'), $text);
514:         }
515: 
516:         // user before handler
517:         $this->invokeHandlers('beforeParse', array($this, & $text, $singleLine));
518: 
519:         // select patterns
520:         $this->_linePatterns = $this->linePatterns;
521:         $this->_blockPatterns = $this->blockPatterns;
522:         foreach ($this->_linePatterns as $name => $foo) {
523:             if (empty($this->allowed[$name])) unset($this->_linePatterns[$name]);
524:         }
525:         foreach ($this->_blockPatterns as $name => $foo) {
526:             if (empty($this->allowed[$name])) unset($this->_blockPatterns[$name]);
527:         }
528: 
529:         // parse Texy! document into internal DOM structure
530:         $this->DOM = TexyHtml::el();
531:         if ($singleLine) {
532:             $this->DOM->parseLine($this, $text);
533:         } else {
534:             $this->DOM->parseBlock($this, $text);
535:         }
536: 
537:         // user after handler
538:         $this->invokeHandlers('afterParse', array($this, $this->DOM, $singleLine));
539: 
540:         // converts internal DOM structure to final HTML code
541:         $html = $this->DOM->toHtml($this);
542: 
543:         // this notice should remain
544:         if (self::$advertisingNotice) {
545:             $html .= "\n<!-- by Texy2! -->";
546:             if (self::$advertisingNotice === 'once') {
547:                 self::$advertisingNotice = FALSE;
548:             }
549:         }
550: 
551:         $this->processing = FALSE;
552: 
553:         return TexyUtf::utf2html($html, $this->encoding);
554:     }

611:     /**
612:      * Converts internal string representation to final HTML code in UTF-8.
613:      * @return string
614:      */
615:     final public function stringToHtml($s)
616:     {
617:         // decode HTML entities to UTF-8
618:         $s = self::unescapeHtml($s);
619: 
620:         // line-postprocessing
621:         $blocks = explode(self::CONTENT_BLOCK, $s);
622:         foreach ($this->postHandlers as $name => $handler) {
623:             if (empty($this->allowed[$name])) continue;
624:             foreach ($blocks as $n => $s) {
625:                 if ($n % 2 === 0 && $s !== '') {
626:                     $blocks[$n] = call_user_func($handler, $s);
627:                 }
628:             }
629:         }
630:         $s = implode(self::CONTENT_BLOCK, $blocks);
631: 
632:         // encode < > &
633:         $s = self::escapeHtml($s);
634: 
635:         // replace protected marks
636:         $s = $this->unProtect($s);
637: 
638:         // wellform and reformat HTML
639:         $this->invokeHandlers('postProcess', array($this, & $s));
640: 
641:         // unfreeze spaces
642:         $s = self::unfreezeSpaces($s);
643: 
644:         return $s;
645:     }

649:     /**
650:      * Converts internal string representation to final HTML code in UTF-8.
651:      * @return string
652:      */
653:     final public function stringToText($s)
654:     {
655:         $save = $this->htmlOutputModule->lineWrap;
656:         $this->htmlOutputModule->lineWrap = FALSE;
657:         $s = $this->stringToHtml( $s );
658:         $this->htmlOutputModule->lineWrap = $save;
659: 
660:         // remove tags
661:         $s = preg_replace('#<(script|style)(.*)</\\1>#Uis', '', $s);
662:         $s = strip_tags($s);
663:         $s = preg_replace('#\n\s*\n\s*\n[\n\s]*\n#', "\n\n", $s);
664: 
665:         // entities -> chars
666:         $s = self::unescapeHtml($s);
667: 
668:         // convert nbsp to normal space and remove shy
669:         $s = strtr($s, array(
670:             "\xC2\xAD" => '',  // shy
671:             "\xC2\xA0" => ' ', // nbsp
672:         ));
673: 
674:         return $s;
